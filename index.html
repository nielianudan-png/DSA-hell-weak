10 Pseudocode Examples

1. Sum of Array: 
START
sum = 0
FOR each element in array
    sum += element
PRINT sum
END
2. Factorial: 
START
result = 1
FOR i = 1 TO n
    result *= i
PRINT result
END
3. Fibonacci: 
START
a = 0, b = 1
FOR i = 1 TO n
    PRINT a
    temp = a + b
    a = b
    b = temp
END
4. Linear Search: 
START
FOR i in array
    IF array[i] == target
        RETURN i
RETURN -1
END
5. Bubble Sort: 
START
FOR i = 0 TO n-1
    FOR j = 0 TO n-i-1
        IF array[j] > array[j+1]
            SWAP array[j] and array[j+1]
END
6. Binary Search: 
START
low = 0, high = n-1
WHILE low <= high
    mid = (low + high)/2
    IF array[mid] == target
        RETURN mid
    ELSE IF array[mid] < target
        low = mid + 1
    ELSE
        high = mid - 1
RETURN -1
END
7. GCD: 
START
WHILE b != 0
    a, b = b, a % b
PRINT a
END
8. Reverse String: 
START
reversed = ''
FOR char FROM end TO start
    append char to reversed
PRINT reversed
END
9. Prime Check: 
START
IF n < 2
    RETURN FALSE
FOR i = 2 TO sqrt(n)
    IF n % i == 0
        RETURN FALSE
RETURN TRUE
END
10. Decimal to Binary: 
START
binary = ''
WHILE n > 0
    binary = n % 2 + binary
    n = n / 2
PRINT binary
END
